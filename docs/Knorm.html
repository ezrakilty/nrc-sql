More Induction Principles on Reduction Sequences

These require knowledge of Continuation.v, so can't be placed in Norm.v Continuations have their own "strong normalization" on their own reduction rule.
Inductive Krw_norm K :=
  reducts_Krw_norm : (∀ K', Krw K K' → Krw_norm K') → Krw_norm K.

Inductive Triple_SN K M N :=
  | triple_sn :
       (∀ K', (Krw K K') → Triple_SN K' M N)
    → (∀ M', (M ~> M') → Triple_SN K M' N)
    → (∀ N', (N ~> N') → Triple_SN K M N')
    → Triple_SN K M N.

Lemma triple_induction_via_TripleSN P:
  (∀ K M N,
        (∀ K', (Krw K K') → P K' M N)
     → (∀ M', (M ~> M') → P K M' N)
     → ((∀ N', (N ~> N') → P K M N'))
     → P K M N)
  →
  ∀ K M N,
    Triple_SN K M N → P K M N.

Lemma triple_induction_via_TripleSN_scoped P:
  ∀ K0 M0 N0,
  (∀ K M N,
     Krw_rt K0 K → (M0 ~>> M) → (N0 ~>> N) →
        (∀ K', (Krw K K') → P K' M N)
     → (∀ M', (M ~> M') → P K M' N)
     → ((∀ N', (N ~> N') → P K M N'))
     → P K M N)
  →
    Triple_SN K0 M0 N0 → P K0 M0 N0.

Lemma Triple_SN_intro:
  ∀ K, Krw_norm K → ∀ M, SN M → ∀ N, SN N → Triple_SN K M N.

Lemma triple_induction_scoped P:
  ∀ K0 M0 N0,
  (∀ K M N,
     Krw_rt K0 K → (M0 ~>> M) → (N0 ~>> N) →
        (∀ K', (Krw K K') → P K' M N)
     → (∀ M', (M ~> M') → P K M' N)
     → ((∀ N', (N ~> N') → P K M N'))
     → P K M N)
  →
    Krw_norm K0 → SN M0 → SN N0 → P K0 M0 N0.

A continuation that is part of, and downstream of, a SN term, is normalizing.
Lemma SN_context_Krw_norm:
  ∀ X X',
    SN X →
    (X ~>> X') →
    ∀ K,
    {M : Term & X' = plug K M} →
    Krw_norm K.

Lemma SN_via_Krw_rt :
  ∀ K K' M,
    Krw_rt K K' → SN (plug K M) → SN (plug K' M).

Lemma SN_via_Krw :
  ∀ K K' M,
    Krw K K' → SN (plug K M) → SN (plug K' M).

Lemma Krw_norm_from_SN:
  ∀ Q, SN Q → ∀ K M, (Q ~>> plug K M) → Krw_norm K.
