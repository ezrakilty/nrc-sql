Library All

Require Import Coq.Lists.List.
Require Import Coq.Lists.ListSet.

Load eztactics.


Definition all A (P : A → Prop) (xs:set A)
  := ∀ (x:A), (set_In x xs) → P x.

Lemma all_nil:
  ∀ A P,
    all A P nil.


Require Import Sets.
Lemma all_map:
  ∀ A eq_dec pred f xs,
    all A pred (set_map eq_dec f xs) ↔ all A (fun x ⇒ pred (f x)) xs.

Lemma all_cons:
  ∀ A P x xs,
    all A P (x::xs) ↔ (P x ∧ all A P xs).

Lemma In_add_intro:
  ∀ A eq_dec (x y:A) xs,
  ({x = y}+{In x xs}) → In x (set_add eq_dec y xs).

Lemma In_add_elim:
  ∀ A eq_dec (x y:A) xs,
    In x (set_add eq_dec y xs) →
    
    (x = y ∨ In x xs).


Lemma all_add_good :
  ∀ A eq_dec P x xs,
    all A P (set_add eq_dec x xs) ↔ P x ∧ all A P xs.

Lemma all_union:
  ∀ A eq_dec pred xs ys,
    all A pred (set_union eq_dec xs ys) ↔
    (all A pred xs ∧ all A pred ys).


Lemma all_map_image:
  ∀ A (p:A → Prop) f xs, (∀ x:A, p (f x)) → all _ p (map f xs).
