Library listkit

List Utilities

Require Import List.

Require Export NthError.

Require Export Foreach.

Require Export All.

Lemma nth_error_foreach_ty :
  ∀ A P v n xs,
    nth_error xs n = value v → foreach_ty A xs P → P v.
Lemma nth_error_In:
  ∀ A xs x (v:A),
    nth_error xs x = value v → ListSet.set_In v xs.

Lemma nth_error_all:
  ∀ A xs x v f,
    nth_error xs x = value v →
    all A f xs →
    f v.
